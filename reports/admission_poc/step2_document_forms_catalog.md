```ts
type StudentDocumentType =
  // 학생 정보 / 신원
  | 'photo'
  | 'passport_copy'
  | 'national_id_copy'                  // 학생 본인
  | 'parents_id_copy'                   // 부모
  | 'alien_registration_copy'           // 국내 체류 시
  // 학력
  | 'highschool_diploma'
  | 'highschool_transcript'
  | 'highschool_diploma_apostille_or_consul'  // 인증 완료본
  // 가족
  | 'birth_certificate'
  | 'family_relations_certificate'      // 베트남: 출생증명 + 호적등본
  // 재정
  | 'bank_balance_certificate'          // 본인 또는 부모 명의
  | 'financial_pledge'                  // 부모 재직증명·사업자등록·자산 증빙 등 부속서류 묶음
  // 학교 소정 양식
  | 'application_form'
  | 'self_intro'                         // 자기소개서
  | 'study_plan'                         // 학업계획서 (분리 또는 통합)
  | 'financial_pledge_form'              // 유학경비 부담 서약서
  | 'privacy_consent'                    // 개인정보 동의서
  | 'academic_record_release'            // 학력 조회 동의서 (옵션)
  // 어학·자격
  | 'topik_certificate'
  | 'language_alt_certificate'           // 세종학당·KIIP·자체 시험 등

  // 비자
  | 'visa_application_form';             // 사증발급인정 신청서
```

## 베트남 특화 처리 룰 (모집요강 종합)

- **학력 인증**: 베트남은 아포스티유 협약 미가입국으로 일반 처리됨 (모집요강마다 명시). **주베트남 한국영사 또는 주한 베트남공관 영사확인** 받은 원본 제출. 베트남이 2025-2026 시점 아포스티유 가입국 여부는 PoC 단계에서 별도 확인 권장
- **가족관계**: "출생증명서 + 호적등본" 조합 (동남보건 명시). 영문 또는 한국어 공증 원본
- **재정 잔고**: 800만원 (군장) / 1,600만원 (대구예술) / 2,000만원 (동남보건) / 1,000만원~ (다른 학교 추정) — **학교 편차 큼**. 화폐 단위는 모두 KRW
- **비자 신청 시 추가**: 베트남 한국대사관은 "지급유보방식의 유학경비 잔고 증명서" 별도 발급 필요 (일반 은행 잔고증명과 다름)
